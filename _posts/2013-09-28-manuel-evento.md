function withCommas(x) {
    return x.toString().replace(/\B(?=(\d{3})+(?!\d))/g, ",");
}

function formatDate(x) {
    var l = x.toString().length;
    if (l === 7) {
        x = '0' + x;
    }
    var d = x.substring(0, 2);
    var m = x.substring(2, 4);
    var y = x.substring(4);
    return m + '/' + d + '/' + y;
}
